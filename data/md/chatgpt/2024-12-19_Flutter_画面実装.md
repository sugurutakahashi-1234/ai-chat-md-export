```dart
class GuildSelectionPage extends StatelessWidget {
  @override
  Widget build(BuildContext context) {

              'Join a guild or create a new one!',
              style: TextStyle(
                fontSize: 16,
                color: Colors.white70,
              ),
              textAlign: TextAlign.center,
            ),
            SizedBox(height: 32),
            GuildOptionButton(
              imagePath: 'Assets.images.animalsLogin.path', // 仮の画像パス
              title: 'Join an existing guild',
              onPressed: () {
                // ボタン押下時の処理
              },
            ),
            SizedBox(height: 16),
            GuildOptionButton(
              imagePath: 'Assets.images.animalsLogin.path', // 仮の画像パス
              title: 'Host a new guild',
              onPressed: () {
                // ボタン押下時の処理
              },
            ),
          ],
        ),
      ),
    );
  }
}

class GuildOptionButton extends StatelessWidget {
  final String imagePath;
  final String title;
  final VoidCallback onPressed;

  GuildOptionButton({
    required this.imagePath,
    required this.title,
    required this.onPressed,
  });

  @override
  Widget build(BuildContext context) {
    return Container(
      decoration: BoxDecoration(
        color: Colors.grey[800], // 仮の背景色
        borderRadius: BorderRadius.circular(16),
      ),
      padding: EdgeInsets.all(16),
      child: Row(
        children: [
          Image.asset(
            imagePath,
            width: 60,
            height: 60,
          ),
          SizedBox(width: 16),
          Expanded(
            child: Text(
              title,
              style: TextStyle(
                fontSize: 18,
                color: Colors.white,
              ),
            ),
          ),
          Icon(
            Icons.arrow_forward,
            color: Colors.white,
          ),
        ],
      ),
    );
  }
}
```

### 概要
1. **`GuildSelectionPage`**: メインの画面で、上部にテキスト、下部に2つのボタンを配置。
2. **`GuildOptionButton`**: 再利用可能なボタンコンポーネント。ボタン内に画像、テキスト、矢印アイコンを含む。

このコードをコピーし、画像パスやスタイルをカスタマイズして仕上げてください！
